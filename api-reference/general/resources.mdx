## GET /general/resources/available-resources

Retrieve a list of all available compute resources including CPUs, GPUs, and storage options with real-time pricing.

### Headers

<ParamField header="Authorization" type="string">
  Optional authentication for personalized pricing
</ParamField>

### Response

<ResponseField name="cpu_instances" type="array">
  Available CPU instance types
  
  <Expandable title="CPU Instance">
    <ResponseField name="instance_type" type="string">
      Instance identifier
    </ResponseField>
    
    <ResponseField name="vcpus" type="integer">
      Number of virtual CPUs
    </ResponseField>
    
    <ResponseField name="memory_gb" type="integer">
      Memory in gigabytes
    </ResponseField>
    
    <ResponseField name="price_per_hour" type="number">
      Hourly price in USD
    </ResponseField>
    
    <ResponseField name="availability" type="string">
      Current availability status
    </ResponseField>
  </Expandable>
</ResponseField>

<ResponseField name="gpu_instances" type="array">
  Available GPU instance types
  
  <Expandable title="GPU Instance">
    <ResponseField name="instance_type" type="string">
      Instance identifier
    </ResponseField>
    
    <ResponseField name="gpu_model" type="string">
      GPU model name
    </ResponseField>
    
    <ResponseField name="gpu_count" type="integer">
      Number of GPUs
    </ResponseField>
    
    <ResponseField name="vram_gb" type="integer">
      GPU memory in gigabytes
    </ResponseField>
    
    <ResponseField name="vcpus" type="integer">
      Number of virtual CPUs
    </ResponseField>
    
    <ResponseField name="memory_gb" type="integer">
      System memory in gigabytes
    </ResponseField>
    
    <ResponseField name="price_per_hour" type="number">
      Hourly price in USD
    </ResponseField>
    
    <ResponseField name="availability" type="string">
      Current availability status
    </ResponseField>
  </Expandable>
</ResponseField>

<ResponseField name="storage_pricing" type="object">
  Storage pricing information
  
  <Expandable title="Storage Pricing">
    <ResponseField name="price_per_gb_month" type="number">
      Monthly storage price per GB
    </ResponseField>
    
    <ResponseField name="price_per_gb_transfer" type="number">
      Data transfer price per GB
    </ResponseField>
  </Expandable>
</ResponseField>

<RequestExample>

```bash cURL
curl -X GET https://api.lyceum.cloud/api/v2/external/general/resources/available-resources \
  -H "Authorization: Bearer YOUR_TOKEN"
```

```python Python
import requests

response = requests.get(
    "https://api.lyceum.cloud/api/v2/external/general/resources/available-resources",
    headers={"Authorization": "Bearer YOUR_TOKEN"}
)
```
